IF (NOT APPLE)
IF (UNIX)

)

    ${QT_QTNETWORK_LIBRARY}
)

    plexydeskcore
    ${libs}
)

    INSTALL(TARGETS plexydeskcore DESTINATION ${CMAKE_INSTALL_PREFIX}/lib/)

    INSTALL(TARGETS plexydeskcore DESTINATION ${CMAKE_INSTALL_PREFIX}/bin/)
